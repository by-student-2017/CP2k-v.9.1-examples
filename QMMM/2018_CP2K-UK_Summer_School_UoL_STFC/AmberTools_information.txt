# Ambertools (GPL) installation (https://ambermd.org/InstUbuntu.php) (Ubuntu)
1. cd $HOME
2. sudo apt update
3. sudo apt -y install wget gfortran gcc cmake libopenmpi-dev
4. tar xvf AmberTools24.tar.bz2
5. cd amber24_src/build
6. ./run_cmake
7. make install
8. AMBER_PREFIX=$(dirname $(dirname `pwd`))
9. echo 'export PATH=$PATH'":${AMBER_PREFIX}/amber24/amber.sh" >> ~/.bashrc
